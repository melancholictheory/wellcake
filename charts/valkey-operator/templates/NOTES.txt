Create a Valkey cluster:
  cat <<EOF | kubectl apply -f -
  apiVersion: cache.wellcake.io/v1alpha1
  kind: ValkeyCluster
  metadata:
    name: example
    namespace: default
  spec:
    topology: Replication
    profile: Cache
    replicas: 3
    resources:
      requests: { cpu: 100m, memory: 256Mi }
      limits:   { cpu: 500m, memory: 1Gi }
  EOF

Status:
  kubectl get valkeyclusters -A
